class Solution {
public:
    int missingNumber(vector<int>& nums) {
        // int sum=0;
        // int n=nums.size();
        // n=(n*(n+1))/2;
        // for(int i=0;i<nums.size();i++)
        // {
        //     sum=sum+nums[i];
        // }
        // return n-sum;
        int n=nums.size();
        int summ=accumulate(nums.begin(),nums.end(),0);
        int t=(n*(n+1))/2;
        return t-summ;
    }
};
