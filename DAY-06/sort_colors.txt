class Solution {
public:
    void sortColors(vector<int>& a) {
        int low=0,mid=0;
        int n=a.size();
        int high=n-1;
        while(mid<=high){
            if(a[mid]==0){
                swap(a[low],a[mid]);
                low++;
                mid++;
            }
            else if(a[mid]==1){
                mid++;
            }
            else if(a[mid]==2){
                swap(a[high],a[mid]);
                high--;
            }
        }
    }
};
