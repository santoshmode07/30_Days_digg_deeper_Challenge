class Solution {
public:
    int pivotIndex(vector<int>& nums) {
        int n=nums.size();
        vector<int>pre(n);
        vector<int>post(n);
        int sum=0,summ=0;
        for(int i=0;i<n;i++){
            pre[i]=sum;
            post[n-i-1]=summ;
            sum+=nums[i];
            summ+=nums[n-i-1];
        }
        for(int i=0;i<n;i++){
            if(pre[i]==post[i]){
                return i;
            }
        }
        return -1;
    }
};
